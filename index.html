STM - succeed & fail


                    object STM {
                      def fail[E](e: E): STM[E, Nothing]
                      def succeed[A](a: A): STM[Nothing, A]
                    }

                    val success = STM.succeed("Hello!")
                    val failure = STM.fail("Nothing to see here...")

STM - chaining


                    trait STM[E, A] {
                      def map[B](f: A => B): STM[E, B]
                      def flatMap[E1 >: E, B](f: A => STM[E1, B]): STM[E1, B]
                    }

                    val sum: STM[Nothing, Long] =
                      for {
                        a <- account1.get
                        b <- account2.get
                      } yield a + b

STM - zipping


                    trait STM[E, A] {
                      def zip[E1 >: E, B](that: => STM[E1, B]): STM[E1, (A, B)]
                    }

                    val bothAccounts: STM[Nothing, (Long, Long)] =
                      account1.get zip account2.get

STM - folds


                    trait STM[E, A] {
                      def fold[B](f: E => B, g: A => B): STM[Nothing, B]

                      def foldM[E1, B](f: E => STM[E1, B],
                                       g: A => STM[E1, B]): STM[E1, B]
                    }

                    val recovered: STM[Nothing, Boolean] =
                      failure.fold(_ => false, _ => true)

STM - check & retry


                    object STM {
                      val retry: STM[Nothing, Nothing]

                      final def check(p: Boolean): STM[Nothing, Unit] =
                        if (p) STM.unit else retry
                    }

STM - alternative


                    trait STM[E, A] {
                      def orElse[E1, A1 >: A](that: => STM[E1, A1]): STM[E1, A1]
                    }

                    STM.fail("failure") orElse STM.succeed(1)

STM - commit


                    trait STM[E, A] {
                      def commit: IO[E, A] = STM.atomically(this)
                    }

                    object STM {
                      def atomically[E, A](stm: STM[E, A]): IO[E, A]
                    }

              def fromEither[E, A](e: => Either[E, A]): STM[E, A]

              def fromFunctionM[R, E, A](f: R => STM[E, A]): ZSTM[R, E, A]
              def fromOption[A](v: => Option[A]): STM[Unit, A]
              def fromTry[A](a: => Try[A]): STM[Throwable, A]
              def left[A](a: => A): STM[Nothing, Either[A, Nothing]]
              def right[A](a: => A): STM[Nothing, Either[Nothing, A]]
              def some[A](a: => A): STM[Nothing, Option[A]]
              def succeed[A](a: => A): STM[Nothing, A]
            }

Mapping


            trait ZSTM[-R, +E, +A] {
              def map[B](f: A => B): ZSTM[R, E, B]
              def mapError[E1](f: E => E1): ZSTM[R, E1, B]
            }

            // STM[Nothing, Int]
            STM.succeed(21).map(_ * 2)

            // STM[Exception, Nothing]
            STM.fail("fail").mapError(msg => new Exception(msg))

Zipping


            trait ZSTM[-R, +E, +A] {
              def zip[R1 <: R, E1 >: E, B](that: => ZSTM[R1, E1, B]): ZSTM[R, E1, (A, B)]
              def zipLeft[R1 <: R, E1 >: E, B](that: => ZSTM[R1, E1, B]): ZSTM[R1, E1, A]
              def zipRight[R1 <: R, E1 >: E, B](that: => ZSTM[R1, E1, B]): ZSTM[R1, E1, B]
              def zipWith[R1 <: R, E1 >: E, B, C](that: => ZSTM[R1, E1, B])(f: (A, B) => C): ZSTM[R1, E1, C]
            }
            
            // STM[Nothing, (Long, Long)]
            account1.get <*> account2.get

Folds


            trait ZSTM[-R, +E, +A] {
              def fold[B](f: E => B, g: A => B): ZSTM[R, Nothing, B]
              def foldM[R1 <: R, E1, B](f: E => ZSTM[R1, E1, B], g: A => ZSTM[R1, E1, B]): ZSTM[R1, E1, B]

            // STM[Nothing, Boolean]
            STM.fail("boom").fold(_ => false, _ => true)
